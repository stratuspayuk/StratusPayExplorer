        // Simulate API call with mock data
        async function simulateAPICall(requestData) {
            // Simulate network delay
            await new Promise(resolve => setTimeout(resolve, 500));

            // Mock calculation logic
            const grossPayPeriod = requestData.annualGrossSalary / getPayPeriodsPerYear(requestData.payFrequency);
            
            // Simple tax calculation (for demo)
            const personalAllowance = 12570;
            const basicRateLimit = 50270;
            let taxableIncome = Math.max(0, requestData.annualGrossSalary - personalAllowance);
            
            if (requestData.isBlindPerson) {
                taxableIncome = Math.max(0, taxableIncome - 2520);
            }

            let annualTax = 0;
            if (requestData.isScottishResident) {
                // Scottish tax bands
                annualTax = calculateScottishTax(taxableIncome);
            } else {
                // Standard UK tax
                const basicRateTax = Math.min(taxableIncome, basicRateLimit - personalAllowance) * 0.20;
                const higherRateTax = Math.max(0, taxableIncome - (basicRateLimit - personalAllowance)) * 0.40;
                annualTax = basicRateTax + higherRateTax;
            }

            const taxPeriod = annualTax / getPayPeriodsPerYear(requestData.payFrequency);

            // NI calculation (simplified)
            const niThreshold = 12570;
            const niableIncome = Math.max(0, requestData.annualGrossSalary - niThreshold);
            const annualNI = Math.min(niableIncome, 50270 - niThreshold) * 0.08 + Math.max(0, niableIncome - (50270 - niThreshold)) * 0.02;
            const niPeriod = annualNI / getPayPeriodsPerYear(requestData.payFrequency);

            // Student loan (simplified)
            let studentLoanPeriod = 0;
            if (requestData.isPlan2) {
                const threshold = 28470;
                const repaymentAmount = Math.max(0, requestData.annualGrossSalary - threshold) * 0.09;
                studentLoanPeriod = repaymentAmount / getPayPeriodsPerYear(requestData.payFrequency);
            }

            // Pension (simplified)
            const employeePensionPeriod = grossPayPeriod * (requestData.employeeContributionPercentage / 100);
            const employerPensionPeriod = grossPayPeriod * (requestData.employerContributionPercentage / 100);

            // Net pay
            const netPayPeriod = grossPayPeriod - taxPeriod - niPeriod - studentLoanPeriod - employeePensionPeriod;

            // Create response object
            const response = {
                uniqueEmployeeId: requestData.uniqueEmployeeId,
                period: requestData.isScottishResident ? "2025-26 (Scottish)" : "2025-26",
                grossPay: grossPayPeriod,
                deductions: {
                    incomeTax: Math.max(0, taxPeriod),
                    nationalInsurance: niPeriod,
                    studentLoanRepayment: studentLoanPeriod,
                    postgraduateLoanRepayment: 0,
                    employeePensionContribution: employeePensionPeriod,
                    employerPensionContribution: employerPensionPeriod,
                    totalDeductions: taxPeriod + niPeriod + studentLoanPeriod + employeePensionPeriod
                },
                netPay: netPayPeriod,
                meta: {
                    calculationTimestamp: new Date().toISOString(),
                    strategyUsed: requestData.isScottishResident ? "TaxStrategy2025 (Scottish)" : "TaxStrategy2025",
                    adjustments: getAppliedAdjustments(requestData)
                }
            };

            displayResults(response);
        }

        function getPayPeriodsPerYear(frequency) {
            const frequencies = {
                'Weekly': 52,
                'FourWeekly': 13,
                'Monthly': 12,
                'Annual': 1
            };
            return frequencies[frequency] || 12;
        }

        function calculateScottishTax(taxableIncome) {
            // Simplified Scottish tax bands
            let tax = 0;
            const bands = [
                { limit: 2306, rate: 0.19 },
                { limit: 13991, rate: 0.20 },
                { limit: 31092, rate: 0.21 },
                { limit: 62732, rate: 0.42 },
                { limit: Infinity, rate: 0.47 }
            ];
            
            let previousLimit = 0;
            for (const band of bands) {
                if (taxableIncome <= previousLimit) break;
                const bandIncome = Math.min(taxableIncome, band.limit) - previousLimit;
                tax += bandIncome * band.rate;
                previousLimit = band.limit;
            }
            return tax;
        }

        function getAppliedAdjustments(requestData) {
            const adjustments = [];
            if (requestData.isScottishResident) adjustments.push("Scottish Tax");
            if (requestData.isBlindPerson) adjustments.push("Blind Person Allowance");
            if (requestData.isMarriageAllowanceTransferor) adjustments.push("Marriage Allowance (Transferor)");
            if (requestData.isMarriageAllowanceRecipient) adjustments.push("Marriage Allowance (Recipient)");
            if (requestData.isPlan1 || requestData.isPlan2 || requestData.isPlan4 || requestData.isPlan5 || requestData.isPostgraduate) {
                adjustments.push("Student Loan");
            }
            if (requestData.employeeContributionPercentage > 0) adjustments.push("Pension Contribution");
            return adjustments.length > 0 ? adjustments : ["Standard Calculation"];
        }

        function displayResults(response) {
            const loading = document.getElementById('loading');
            const resultsSection = document.getElementById('resultsSection');

            // Update summary
            document.getElementById('grossPayValue').textContent = formatCurrency(response.grossPay);
            document.getElementById('totalDeductionsValue').textContent = formatCurrency(response.deductions.totalDeductions);
            document.getElementById('netPayValue').textContent = formatCurrency(response.netPay);

            // Update deductions
            document.getElementById('incomeTaxValue').textContent = formatCurrency(response.deductions.incomeTax);
            document.getElementById('niValue').textContent = formatCurrency(response.deductions.nationalInsurance);
            document.getElementById('studentLoanValue').textContent = formatCurrency(response.deductions.studentLoanRepayment);
            document.getElementById('postgraduateLoanValue').textContent = formatCurrency(response.deductions.postgraduateLoanRepayment);
            document.getElementById('employeePensionValue').textContent = formatCurrency(response.deductions.employeePensionContribution);

            // Update deduction bar
            const total = response.grossPay;
            const segments = [
                { value: response.deductions.incomeTax, class: 'bar-income-tax', label: 'Tax' },
                { value: response.deductions.nationalInsurance, class: 'bar-ni', label: 'NI' },
                { value: response.deductions.studentLoanRepayment + response.deductions.postgraduateLoanRepayment, class: 'bar-student-loan', label: 'Loans' },
                { value: response.deductions.employeePensionContribution, class: 'bar-pension', label: 'Pension' },
                { value: response.netPay, class: 'bar-net', label: 'Net' }
            ];

            const barHTML = segments
                .filter(s => s.value > 0)
                .map(s => {
                    const percentage = (s.value / total * 100).toFixed(1);
                    return `<div class="bar-segment ${s.class}" style="flex: ${percentage}%;" title="${s.label}: ${formatCurrency(s.value)}">${percentage}%</div>`;
                })
                .join('');

            document.getElementById('deductionBar').innerHTML = barHTML;

            // Update metadata
            document.getElementById('periodValue').textContent = response.period;
            document.getElementById('strategyValue').textContent = response.meta.strategyUsed;
            document.getElementById('timestampValue').textContent = new Date(response.meta.calculationTimestamp).toLocaleTimeString('en-GB');

            // Show results
            loading.classList.remove('active');
            resultsSection.classList.add('active');
        }
